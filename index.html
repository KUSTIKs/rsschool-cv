    function dblLinear(n) {
  const res = [1];
  let x1 = 0,
      x2 = 0;
  while(res.length <= n) {
    y = 2 * res[x1] + 1;
    z = 3 * res[x2] + 1;
    if (y < z){
      res.push(y);
      x1++;
    } else if (y > z) {
      res.push(z);
      x2++;
    } else {
      res.push(y);
      x1++;
      x2++;
    }
  }
  return res[n];
}

    def tour(friends, friend_towns, home_to_town_distances):
    
  friend_towns = [a for a in friend_towns if a[0] in friends]
  friend_towns.sort(key = lambda a: friends.index(a[0]))
  
  keys = [a[1] for a in friend_towns]
  res = home_to_town_distances[keys[0]]
  prev_dis = res
  
  for key in keys[1:]: 
    dis = home_to_town_distances[key]
    res += (dis ** 2 - prev_dis ** 2) ** 0.5
    prev_dis = dis
      
  res += home_to_town_distances[keys[-1]]
    
return int(res)
